$.isObject returns true or false if the passed in parameter is indeed an object.

See the following examples


```
var notObject=function(){};
var isObject={};

function testisObject(pass){
	if(pass)
		alert($.isObject(isObject));
	else
		alert($.isObject(notObject));
}
```
